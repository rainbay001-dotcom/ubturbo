1. Design Overview

Option A implements NVMe swap as a custom kernel-space swap engine inside the SMAP kernel module. Instead of relying on the kernel’s existing swap subsystem (swapon, swap_cache, frontswap), Option A directly manages swap slot allocation, PTE manipulation, and block I/O via submit_bio().

Key Advantage: Option A can swap 2M hugepages without splitting them into 4K base pages (no THP splitting), preserving hugepage benefits on swap-in. This is impossible with the standard kernel swap path.

3. Swap Map Allocator

The swap map allocator (smap_swap_map.c) manages NVMe swap slot allocation using a kernel bitmap. It supports two slot sizes: 4K (standard pages) and 2M (hugepages).

3.1 Bitmap Design

Each NVMe swap device is divided into fixed-size slots. A bitmap tracks allocation state — one bit per slot. For a 128 GB NVMe partition with 4K slots, this requires ~4 MB of bitmap memory.

struct smap_swap_map {
    unsigned long *bitmap;       /* allocation bitmap, 1 bit per slot */
    uint32_t      total_slots;   /* total number of slots */
    uint32_t      slot_size;     /* PAGE_SIZE (4096) or PMD_SIZE (2M) */
    atomic_t      free_slots;    /* fast free count */
    spinlock_t    lock;          /* protects bitmap modifications */
    unsigned long hint;          /* last allocation position (search hint) */
};

3.2 4K and 2M Slot Sizes

The allocator maintains two separate bitmaps — one for 4K slots and one for 2M slots. The 2M bitmap manages contiguous 512-slot regions. When a 2M hugepage needs to be swapped out, a single bit in the 2M bitmap is set, reserving 512 contiguous 4K-aligned slots on NVMe.

Design choice: Separate bitmaps avoid fragmentation between 4K and 2M allocations. The NVMe partition is divided at initialization: e.g., 80% for 4K slots, 20% for 2M slots.

3.3 Allocation & Free

/* Allocate a swap slot. Returns slot index or -ENOSPC. */
int smap_swap_alloc_slot(struct smap_swap_map *map)
{
    unsigned long idx;
    int ret = -ENOSPC;

    spin_lock(&map->lock);
    idx = find_next_zero_bit(map->bitmap, map->total_slots, map->hint);
    if (idx >= map->total_slots)
        idx = find_first_zero_bit(map->bitmap, map->total_slots);
    if (idx < map->total_slots) {
        set_bit(idx, map->bitmap);
        map->hint = idx + 1;
        atomic_dec(&map->free_slots);
        ret = idx;
    }
    spin_unlock(&map->lock);
    return ret;
}

/* Free a swap slot. */
void smap_swap_free_slot(struct smap_swap_map *map, unsigned long idx)
{
    BUG_ON(idx >= map->total_slots);
    BUG_ON(!test_bit(idx, map->bitmap));
    clear_bit(idx, map->bitmap);
    atomic_inc(&map->free_slots);
}

/* Initialize the swap map for a device. */
struct smap_swap_map *smap_swap_map_create(uint64_t capacity_bytes,
                                           uint32_t slot_size)
{
    struct smap_swap_map *map;
    uint32_t total;

    map = kzalloc(sizeof(*map), GFP_KERNEL);
    if (!map)
        return ERR_PTR(-ENOMEM);

    total = capacity_bytes / slot_size;
    map->bitmap = bitmap_zalloc(total, GFP_KERNEL);
    if (!map->bitmap) {
        kfree(map);
        return ERR_PTR(-ENOMEM);
    }

    map->total_slots = total;
    map->slot_size   = slot_size;
    atomic_set(&map->free_slots, total);
    spin_lock_init(&map->lock);
    map->hint = 0;

    return map;
}

4. Block I/O Engine

The block I/O engine (smap_swap_bio.c) provides asynchronous page read/write to NVMe via the kernel’s submit_bio() interface.

4.1 submit_bio Mechanics

struct smap_swap_bio_ctx {
    struct page       *page;       /* page being transferred */
    unsigned long      slot_idx;   /* NVMe swap slot index */
    struct completion   done;       /* completion for sync wait */
    int                error;      /* bio error code */
    bool               is_write;   /* direction */
};

/* Submit a single page bio to NVMe. */
int smap_swap_bio_submit(struct smap_swap_device *dev,
                        struct page *page,
                        unsigned long slot_idx,
                        enum req_op op,
                        struct smap_swap_bio_ctx *ctx)
{
    struct bio *bio;
    sector_t sector;

    sector = (sector_t)slot_idx * (dev->map->slot_size / 512);

    bio = bio_alloc(dev->bdev, 1, op, GFP_NOIO);
    if (!bio)
        return -ENOMEM;

    bio->bi_iter.bi_sector = sector;
    bio_add_page(bio, page, dev->map->slot_size, 0);
    bio->bi_end_io = smap_swap_bio_end;
    bio->bi_private = ctx;

    ctx->page     = page;
    ctx->slot_idx = slot_idx;
    ctx->error    = 0;
    ctx->is_write = (op == REQ_OP_WRITE);
    init_completion(&ctx->done);

    submit_bio(bio);
    return 0;
}

4.2 Completion Callbacks

static void smap_swap_bio_end(struct bio *bio)
{
    struct smap_swap_bio_ctx *ctx = bio->bi_private;

    ctx->error = blk_status_to_errno(bio->bi_status);
    if (ctx->error)
        pr_warn("smap_swap: bio %s error %d slot %lu\n",
                ctx->is_write ? "write" : "read",
                ctx->error, ctx->slot_idx);

    complete(&ctx->done);
    bio_put(bio);
}

4.3 Batched I/O

For swap-out of multiple pages, the engine supports batched submission using bio_list. This amortizes per-bio overhead and allows the NVMe device to optimize its internal scheduling.

/* Submit a batch of pages for write. Returns number successfully submitted. */
int smap_swap_bio_batch_write(struct smap_swap_device *dev,
                              struct page **pages,
                              unsigned long *slots,
                              int count,
                              struct smap_swap_bio_ctx *ctxs)
{
    int i, submitted = 0;

    for (i = 0; i < count; i++) {
        int ret = smap_swap_bio_submit(dev, pages[i], slots[i],
                                       REQ_OP_WRITE, &ctxs[i]);
        if (ret == 0)
            submitted++;
        else
            ctxs[i].error = ret;
    }
    return submitted;
}

/* Wait for all bios in batch to complete. */
int smap_swap_bio_batch_wait(struct smap_swap_bio_ctx *ctxs,
                             int count)
{
    int i, errors = 0;
    for (i = 0; i < count; i++) {
        wait_for_completion(&ctxs[i].done);
        if (ctxs[i].error)
            errors++;
    }
    return errors;
}

4.4 Error Handling & Retry

Retry policy: NVMe I/O errors are retried up to 3 times with exponential backoff (1ms, 4ms, 16ms). If all retries fail, the page is marked as failed in the result array and the swap slot is freed. On read failure during swap-in, the process receives SIGBUS.

5. Swap-Out Path

The swap-out path moves cold pages from L2 (CXL) to L3 (NVMe). It is triggered by user-space via ioctl(SMAP_SWAP_OUT) and executes entirely in kernel context.

Step 10: Update Accounting

After all bios complete (or fail), update atomic64_t pages_swapped_out and write per-page results to swap_msg.results[] via copy_to_user().

6. Swap-In / Page Fault Path

When user-space accesses a swapped-out page, the CPU triggers a page fault. The kernel’s do_swap_page() sees a swap PTE. A kprobe intercepts the call to handle SMAP swap entries.

7. PTE Swap Entry Encoding

On aarch64, a non-present PTE has bits [63:2] available for software use. SMAP uses a custom swap type ID to distinguish its entries from kernel-managed swap.

aarch64 Swap PTE Layout

 63       58 57                                   2  1  0
+----------+--------------------------------------+--+--+
| SWP_TYPE |          SWP_OFFSET                   |  | 0|
|  (6 bits)|          (56 bits)                    |  |  |
+----------+--------------------------------------+--+--+
                                                    ^
                                                    |
                                              Present=0 (swap)

Encoding Functions

#define SMAP_SWAP_TYPE    30   /* custom type, avoids kernel swap types 0-5 */
#define SMAP_HUGE_FLAG    (1UL << 55)   /* bit 55 = hugepage indicator */

static inline swp_entry_t smap_encode_swap_entry(unsigned long slot_idx,
                                                  bool is_huge)
{
    unsigned long offset = slot_idx;
    if (is_huge)
        offset |= SMAP_HUGE_FLAG;
    return swp_entry(SMAP_SWAP_TYPE, offset);
}

static inline void smap_decode_swap_entry(swp_entry_t entry,
                                           unsigned long *slot_idx,
                                           bool *is_huge)
{
    unsigned long offset = swp_offset(entry);
    *is_huge  = !!(offset & SMAP_HUGE_FLAG);
    *slot_idx = offset & ~SMAP_HUGE_FLAG;
}
Type ID selection: SMAP_SWAP_TYPE = 30 is chosen to avoid collision with kernel swap types (typically 0–5 for swap partitions/files). The kprobe on do_swap_page checks swp_type(entry) == SMAP_SWAP_TYPE to intercept only SMAP entries.

Hugepage Encoding Detail

For 2M hugepages, the swap entry stores the base slot index (the first of 512 contiguous slots) with bit 55 set. On swap-in, the fault handler reads bit 55, then reads all 512 slots starting from the base index to reconstruct the full 2M page.

8. 2M Hugepage Swap (Option A Key Advantage)

This is Option A’s primary advantage over Option B. The standard kernel swap path (MADV_PAGEOUT) calls split_huge_page() before swapping, destroying THP mappings. Option A preserves hugepage granularity end-to-end.

Swap-Out: 2M Page

  1. Allocate one 2M slot from the 2M bitmap (equivalent to 512 contiguous 4K slots on NVMe)
  2. Encode swap entry with SMAP_HUGE_FLAG set and base slot index
  3. Write full 2M as a single large bio: bio_add_page(bio, page, PMD_SIZE, 0)
  4. Or split into 512 individual 4K bios for devices with smaller max request size
  5. Install swap PMD entry via set_pmd_at()

Swap-In: 2M Page

  1. Fault handler detects SMAP_HUGE_FLAG in swap entry
  2. Attempt: alloc_pages_node(nid, GFP_HIGHUSER_MOVABLE | __GFP_COMP, 9) — order-9 = 2M
  3. If hugepage allocation succeeds: read all 512 slots into the compound page
  4. Install PMD entry — no THP splitting, full hugepage restored
  5. Fallback: if hugepage unavailable (fragmented), split read into 512 individual 4K pages
/* Swap-in a 2M hugepage from NVMe. */
static int smap_swap_in_huge(struct mm_struct *mm,
                              unsigned long addr,
                              unsigned long base_slot,
                              int nid)
{
    struct page *hpage;
    int i, ret;

    /* Try to allocate a 2M compound page on the faulting NUMA node */
    hpage = alloc_pages_node(nid,
        GFP_HIGHUSER_MOVABLE | __GFP_COMP | __GFP_NOWARN, 9);

    if (!hpage) {
        /* Fallback: split into 512 x 4K reads */
        return smap_swap_in_split(mm, addr, base_slot, nid);
    }

    /* Submit 512 bio reads for contiguous NVMe slots */
    for (i = 0; i < 512; i++) {
        struct page *subpage = hpage + i;
        ret = smap_swap_bio_submit(dev, subpage,
                                    base_slot + i,
                                    REQ_OP_READ, &ctxs[i]);
        if (ret)
            goto err_free;
    }

    /* Wait for all 512 reads */
    smap_swap_bio_batch_wait(ctxs, 512);

    /* Install PMD entry */
    set_pmd_at(mm, addr & PMD_MASK,
               pmd_offset(pud, addr),
               mk_pmd(hpage, vma->vm_page_prot));

    /* Free all 512 swap slots */
    smap_swap_free_slot(&dev->map_2m, base_slot / 512);

    return 0;

err_free:
    __free_pages(hpage, 9);
    return -EIO;
}

/* Option A: swap out via ioctl (not process_madvise) */
int DoSwapOut(struct ManageInfo *info, struct PageBatch *batch)
{
    struct swap_msg msg = {
        .cnt   = batch->count,
        .pid   = batch->pid,
        .addrs = batch->phys_addrs,
    };
    int ret;

    msg.results = calloc(batch->count, sizeof(uint64_t));
    if (!msg.results)
        return -ENOMEM;

    ret = ioctl(info->swap_fd, SMAP_SWAP_OUT, &msg);

    /* Check per-page results */
    for (int i = 0; i < batch->count; i++) {
        if (msg.results[i] < 0)
            swap_account_error(info->acct, msg.results[i]);
        else
            swap_account_out(info->acct, 1);
    }

    free(msg.results);
    return ret;
}

14. Fault Hook Strategy

The swap-in path requires intercepting page faults for SMAP swap entries. Two approaches are considered:

Approach A: Register Custom swap_type

  • Register SMAP_SWAP_TYPE with the kernel swap subsystem via swap_info_struct
  • Provide custom swap_readpage() and swap_writepage() ops
  • Pro: Cleanest integration, kernel calls our readpage automatically
  • Con: Requires kernel source patch to mm/swap_state.c — not possible with out-of-tree module

Approach B: Kprobe on do_swap_page (Recommended)

  • Register a kprobe on do_swap_page() in mm/memory.c
  • In the pre-handler, check if swp_type(entry) == SMAP_SWAP_TYPE
  • If yes: handle the fault ourselves, set return value, skip original function
  • If no: return 0 to let kernel handle normally
  • Pro: Consistent with existing smap_migrate_wrapper.c kprobe pattern
  • Con: Fragile across kernel versions (function signature may change)
Recommendation: Use Approach B (kprobe) for consistency with the existing SMAP codebase. The smap_migrate_wrapper.c module already uses kprobes to intercept kernel migration paths, so the pattern is well-established and tested.

Kprobe Implementation Sketch

static int smap_swap_fault_pre(struct kprobe *p, struct pt_regs *regs)
{
    struct vm_fault *vmf = (struct vm_fault *)regs->regs[0];
    swp_entry_t entry;
    unsigned long slot_idx;
    bool is_huge;

    entry = pte_to_swp_entry(vmf->orig_pte);
    if (swp_type(entry) != SMAP_SWAP_TYPE)
        return 0;  /* not ours, let kernel handle */

    smap_decode_swap_entry(entry, &slot_idx, &is_huge);

    if (is_huge)
        vmf->ret = smap_swap_in_huge(vmf->vma->vm_mm,
                                      vmf->address, slot_idx,
                                      numa_node_id());
    else
        vmf->ret = smap_swap_in_page(vmf->vma->vm_mm,
                                      vmf->address, slot_idx,
                                      numa_node_id());

    /* Skip the original do_swap_page */
    instruction_pointer_set(regs, ((unsigned long)p->addr) + p->size);
    return 1;
}

  • Hugepage swap-in: compound page alloc, batch read, PMD install
  • Fallback path: split to 4K if hugepage unavailable
  • Test: swap 2M page, verify hugepage preserved on swap-in
Phase 5: User-Space Integration (1 week)
  • Integrate cold_tracker with swap-out path
  • Implement DoSwapOut() in migration.c using ioctl
  • Add swap stats to monitoring output
  • Add SMAP_SWAP_STATS ioctl and user-space reader
  • Config: NVMe device path, 4K/2M split ratio
Phase 6: Testing, Debugging, Performance Tuning (1.5 weeks)
  • Stress test: concurrent swap-out and fault-driven swap-in
  • Correctness: verify data integrity after swap round-trip
  • Performance: measure swap-out throughput, swap-in latency (P50/P99)
  • Edge cases: out-of-space, NVMe errors, process exit during swap
  • Memory leak check: verify all pages freed, all slots returned
  • Compare vs Option B: latency, throughput, hugepage preservation
Total estimated effort: ~7–8 weeks for a single developer. Phase 3 (kprobe fault handler) and Phase 4 (2M hugepage) are the highest-risk phases due to direct PTE manipulation in fault context.
